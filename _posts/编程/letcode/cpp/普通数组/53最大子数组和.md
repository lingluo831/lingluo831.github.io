题:给你一个整数数组 `nums` ，请你找出一个具有最大和的连续子数组（子数组最少包含一个元素），返回其最大和。
是数组中的一个连续部分。

**思路**
利用前缀和来遍历所有数组,前缀和只能从一个数开始逐次增加相加的元素,所以需要另一层遍历来移动初始值的位置

```cpp
class Solution {
public:
    int maxSubArray(vector<int>& nums) 
    {
        int n = nums.size();
        vector<int> prefix(n);
        prefix[0] = nums[0];
        for (int i = 1; i < n; ++i) 
        {
            prefix[i] = prefix[i - 1] + nums[i];
        }
        int max_sum = nums[0];
        for (int i = 0; i < n; ++i) 
        {
            for (int j = i; j < n; ++j) 
            {
                int sum;
                if (i == 0) 
                {
                    sum = prefix[j];
                } else 
                {
                    sum = prefix[j] - prefix[i - 1];
                }
                max_sum = max(max_sum, sum);
            }
        }
        return max_sum;
    }
};
```
这个方法超时了.

**答案1** [[编程/思路/动态规划|动态规划]]

```cpp
class Solution {
public:
    int maxSubArray(vector<int> &nums) 
    {
        int result = INT_MIN; // 初始最大值设为最小整数，保证任何子数组和都能更新它
        int numsSize = int(nums.size());        
        vector<int> dp(numsSize);// dp[i]表示以 nums[i] 结尾的最大子序和
        dp[0] = nums[0];      // 以第一个元素结尾的最大和就是第一个元素本身
        result = dp[0];       // 更新最大和
        for (int i = 1; i < numsSize; i++) 
        {            
            dp[i] = max(dp[i - 1] + nums[i], nums[i]);// 递推公式：要么把 nums[i] 加到前面的最大和里，要么就自己单独重新开始
            result = max(result, dp[i]); // 更新最大值
        }
        return result;
    }
};
```

答案2 贪心算法

```cpp
class Solution {
public:
    int maxSubArray(vector<int> &nums) {
        // 类似寻找最大最小值的题目，初始值一定要定义成理论上的最小值
        int result = INT_MIN;
        int numsSize = int(nums.size());
        int sum = 0;
        for (int i = 0; i < numsSize; i++) {
            sum += nums[i];
            result = max(result, sum);
            // 如果sum < 0，重新开始找子序串
            if (sum < 0) {
                sum = 0;
            }
        }
        return result;
    }
};
```
